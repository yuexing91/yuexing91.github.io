# 全局API

import { Configurator, createVucNode } from 'vuc3'

# Configurator

Vuc的配置对象。

# registerConfig

注册配置方法

# setVucConfig

设置组件配置

Configurator.setVucConfig({
  id: 'demo-button',
  props: {
    text: {
      label: '按钮标签',
      editors: 'string',
    },
  },
})

# getVucConfig

获取组件配置

Configurator.getVucConfig('demo-button')

# setVucProxyCreateApp

设置创建Vue的代理组件被创建时的处理逻辑。

Configurator.setVucProxyCreateApp((VucProxy) => {
  let proxyApp = createApp(VucProxy);
  proxyApp.use(Antd);
  return proxyApp;
});

# setVucProxyHooks

设置 VucProxy 代理组件的钩子方法。

Configurator.setVucEditorHooks('onBeforeRender', ({ vucAst }) => {

})
  • onCreatedApp 通过 vue.createApp 创建完 app 之后
  • onBeforeRender 开始渲染之前
  • onRender 单个节点渲染时

# setVucEditorHooks

设置 Vuc 编辑器的钩子方法。

Configurator.setVucEditorHooks('onBeforeSelectNode', (vucNode) => {
  return false;
})
  • onBeforeSelectNode 选中节点之前，返回false取消选中
  • onDragdrop 拖拽放置时，返回false禁止放置

# registerAstProcess

注册 createVucAst 时的处理逻辑。

# installModule

安装 ES6 模块

# registerValueEditor

注册 内容编辑器

Configurator.registerValueEditor('col', {
  name: '栅格',
  dataType: 'object',
  Editor: ColEditor,
});

# registerComboType

注册一个组合式组件。

Configurator.registerComboType({
  id: 'FormItem',
  name: '表单项',
  template: `<a-form-item key="formItem">
      <type.input key="input" allowChildren="true"/>
    </a-form-item>`,
  configs() {
    return [
      {
        title: '属性',
        mode: 'prop',
        items: [
          {
            nodeKey: 'formItem',
            name: 'name',
            label: '字段名',
          },
          {
            nodeKey: 'input',
            name: 'size',
            label: '尺寸',
          },
        ],
      },
    ];
  },
});

# registerComboMode

注册一个组合式模型

Configurator.registerComboMode({
  mode: 'prop',
  Component: PropPanel,
  parseComponentProps(config, combo) {

  },
})

# createVucAst

返回一个 VucAst 对象，它是用来描述 Vue SFC 的对象。

createVucAst(`<template>
</template>
<script>
export default {
};
</script>`)

# createVucNode

返回一个 VucNode 对象，它用来描述 template 对应的节点信息。

createVucNode('<div></div>')
