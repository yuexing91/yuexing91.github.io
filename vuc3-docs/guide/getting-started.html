# 起步

# 安装

$ npm install vuc3 --save

# 使用

  • 导入模块

import 'vuc3/vuc.css';
import { Configurator, SampleDesigner } from 'vuc3';

// 依赖 ant-design-vue 样式
import 'ant-design-vue/dist/antd.css';
  • 注册组件配置

Configurator.setVucConfig({
  id: 'a-button',
  name: '按钮',
  ...// 其他配置
});
  • 创建vuc3设计器
let app = createApp(SampleDesigner, {
  vueContent: `<template>
  <a-card></a-card>  
</template>
<script>
export default {
};
</script>`,
});

app.mount('#app');

# 示例

main.js


// 依赖 ant-design-vue
import 'ant-design-vue/dist/antd.css';
import Antd from 'ant-design-vue';

import 'vuc3/vuc.css';
import { Configurator, SampleDesigner } from 'vuc3';

import { createApp, h } from 'vue';

// 组件的配置文件
import ButtonConfig from './configs/components/ButtonConfig';
import CardConfig from './configs/components/CardConfig';
import InputConfig from './configs/components/InputConfig';

// 注册组件的配置
Configurator.setVucConfig(ButtonConfig);
Configurator.setVucConfig(CardConfig);
Configurator.setVucConfig(InputConfig);

/**
 * 设置创建代理的vue应用，在这里可以为需要编辑的sfc文件注入需要的内容
 * 例如本示例需要使用ant-design-vue
 */
Configurator.setVucProxyCreateApp((VucProxy) => {
  let proxyApp = createApp(VucProxy);
  proxyApp.use(Antd);
  return proxyApp;
});

// 需要编辑的vue文件内容
let vueContent = `<template>
</template>
<script>
export default {
};
</script>`;

// 组件资源管理器的配置
let explorerComponents = [
  {
    title: '基础',
    children: [
      {
        title: '按钮',
        tag: 'a-button',
        node: `<a-button>按钮</a-button>`,
      },
    ],
  },
  {
    title: '布局',
    children: [
      {
        title: '卡片',
        tag: 'a-card',
        node: '<a-card title="卡片"></a-card>',
      },
    ],
  },
  {
    title: '表单',
    children: [
      {
        title: '输入框',
        tag: 'a-input',
        node: `<a-input></a-input>`,
      },
    ],
  },
];

// 创建设计器
let app = createApp(SampleDesigner, {
  vueContent,
  explorerComponents,
  style: 'height:100%',
});

app.mount('#app');

ButtonConfig.js

export default {
  // 唯一ID，对应标签名称
  id: 'a-button',
  // 显示名称
  name: '按钮',
  // 事件配置
  eventTypes: [
    {
      // 事件的类型
      id: 'click',
      // 事件显示名称
      name: '点击',
    },
  ],
  // 插槽配置
  slots: [
    {
      slot: 'default',
      name: '按钮内容',
    },
    {
      slot: 'icon',
      name: '图标',
    },
  ],
  // 组件类型
  type: ['button'],
  // 组件props配置 
  props: {
    type: {
      label: '按钮类型',
      editors: 'select?@items=default|primary|dashed|text|danger|link',
    },
    disabled: {
      label: '禁用',
      editors: 'boolean',
    },
    size: {
      label: '尺寸',
      editors: {
        name: '选项',
        type: 'radio-group',
        props: {
          items: [
            {
              value: 'large',
              label: '大',
            },
            {
              value: 'default',
              label: '中',
            },
            {
              value: 'small',
              label: '小',
            },
          ],
        },
      },
    },
  },
  // 文本节点配置
  text: {
    label: '按钮文本',
  },

  // 拖拽时子节点限制，返回false不允许有子节点
  childLimit() {
    return false;
  },

  // 代理配置
  vucProxyOption: {
    textProxy: false,
  },
};
